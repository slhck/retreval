Retreval::QueryResult

A QueryResult contains a list of results for a given query. It can either be ranked or unranked. You can’t instantiate such a class - use the subclasses RankedQueryResult and UnrankedQueryResult instead.

f_measure(args = {:beta => 1}) click to toggle source

Calculates the F-measure, weighing precision and recall. See: nlp.stanford.edu/IR-book/html/htmledition/evaluation-of-unranked-retrieval-sets-1.html 

     # File lib/retreval/query_result.rb, line 188
188:     def f_measure(args = {:beta => 1})
189:       
190:       # Get intermediate results for both un/ranked results
191:       results = calculate
192:       
193:       precision = results[:precision]
194:       recall = results[:recall]
195:       
196:       begin
197:         # The user has the option to supply either alpha or beta (or both, doesn't matter)
198:         unless args[:alpha].nil?
199:           alpha = args[:alpha].to_f
200:           beta_squared = (1 - alpha) / alpha
201:         end
202:         
203:         unless args[:beta].nil?
204:           beta = args[:beta].to_f
205:           beta_squared = beta * beta
206:         end
207:         
208:         ((beta_squared + 1) * precision * recall) / ((beta_squared * precision) + recall)
209:         
210:       rescue Exception => e
211:         raise "Error while calculating F-Measure: " + e.message
212:       end
213:       
214:     end

calculate(resultset = nil) click to toggle source

This is the method that performs all necessary calculations on a set of results. Never call this. It should be called automatically.

228:     def calculate(resultset = nil)
229:       
230:       # Use the gold standard we initially received
231:       standard = @gold_standard
232:       
233:       # If there is an unranked result to be calculated ..
234:       if resultset.nil?
235:         unranked = true
236:         # ... we'll just take the array of documents for this result
237:         resultset = OpenStruct.new
238:         resultset.documents = @documents.values
239:         resultset.query = @query
240:       end
241: 
242:       begin
243:         all_items = standard.documents.length               # => all documents this gold standard contains
244:         retrieved_items = resultset.documents.length        # => all items retrieved for this information need
245:         not_retrieved_items = all_items - retrieved_items   # => all items NOT retrieved for this information need
246:         retrieved_relevant_items = 0                        # => the count of retrieved and relevant documents
247:         not_retrieved_relevant_items = 0                    # => the count of nonretrieved relevant documents
248: 
249:         # Get the query we are working on
250:         query = resultset.query
251:       
252:         # Get the document sets we are working on
253:         retrieved_documents = resultset.documents           # => This is an Array all the time
254:         # This means that we will filter out all retreived documents from the Gold Standard
255:         # TODO: This can definitely be improved, but how?
256:         not_retrieved_documents = standard.documents.reject { |key, doc| retrieved_documents.include? doc } # => This is a Hash as it refers to the Gold Standard
257:         
258:         # Check whether each of the retrieved documents is relevant or not ...
259:         retrieved_documents.each do |doc|
260:           relevant = standard.relevant? :document => doc.id, :query => query.querystring
261:           retrieved_relevant_items +=1 if relevant
262:         end
263:         retrieved_nonrelevant_items = retrieved_items - retrieved_relevant_items
264:         
265:         # ... do the same for nonretrieved documents. This is a hash, so we only take the values
266:         not_retrieved_documents.values.each do |doc|
267:           relevant = standard.relevant? :document => doc.id, :query => query.querystring
268:           not_retrieved_relevant_items += 1 if relevant
269:         end
270:         not_retrieved_nonrelevant_items = not_retrieved_items - not_retrieved_relevant_items
271:       
272:         # Now calculate the sum counts
273:         relevant_items = retrieved_relevant_items + not_retrieved_relevant_items
274:         nonrelevant_items = retrieved_nonrelevant_items + not_retrieved_nonrelevant_items
275:       
276:         # Finally, calculate precision and recall
277:         precision = retrieved_relevant_items.to_f / retrieved_items.to_f
278:         if relevant_items != 0
279:           recall = retrieved_relevant_items.to_f / relevant_items.to_f
280:         else
281:           recall = 0
282:         end
283:         
284:         # This hash will be available as the result later
285:         results = {
286:           :precision => precision,
287:           :recall => recall,
288:           :false_negatives => not_retrieved_relevant_items,
289:           :false_positives => retrieved_nonrelevant_items,
290:           :true_negatives => not_retrieved_nonrelevant_items,
291:           :true_positives => retrieved_relevant_items
292:         }
293:

295:         # newly added document is relevant (thus increasing precision and recall)

298:           results[:relevant] = standard.relevant? :document => results[:document], :query => query.querystring
