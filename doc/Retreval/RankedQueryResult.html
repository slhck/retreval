Retreval::RankedQueryResult

A RankedQueryResult is a QueryResult with special functions for ranked retrieval evaluation.

Public Class Methods

new(args) click to toggle source

Creates a new RankedQueryResult. One has to specify the query string and can optionally pass a Document array too. The rank of the Document will be defined by its position in the array only. Called by:

eleven_point_precision() click to toggle source

Calculates the 11-point precision and the average interpolated precision. See: nlp.stanford.edu/IR-book/html/htmledition/evaluation-of-ranked-retrieval-results-1.html 

     # File lib/retreval/query_result.rb, line 326
326:     def eleven_point_precision
327:       
328:       statistics unless @calculated
329: 
330:       @recall_levels = Hash.new
331: 
332:       # Find out what recall/precision pairs we already know
333:       @results.each_with_index do |row, index|
334:         precision = row[:precision]
335:         recall = row[:recall]
336:         @recall_levels[recall] = precision
337:       end
338: 
339:       begin
340:         @eleven_point_precision = Hash.new
341: 
342:         # Calculate the 11 points
343:         # This outer loop effectively iterates from 0.0 to 1.0
344:         (0..10).each_with_index do |recall_level, index|
345:           recall_level = recall_level.to_f / 10
346:           @eleven_point_precision[recall_level] = 0
347: 
348:           # Look in our known recall levels (stored as the keys)
349:           @recall_levels.keys.each do |key|
350:             
351:             # If we find a known recall equal or higher to the one from the 11-point
352:             # scale, return the precision at that level
353:             # E.g. if our level is 0.3 and we have a known precision at 0.5, this
354:             # one will be used as our interpolated precision
355:             if key >= recall_level
356:               precision_for_level = @recall_levels[key]
357:               # Store the interpolated precision at the current level, e.g. 0.3
358:               @eleven_point_precision[recall_level] = precision_for_level
359:               break
360:             end
361:           end
362:         end
363: 
364:         # Now calculate an average precision for this statistic
365:         # That's a neat line of ruby code, is it?
366:         @eleven_point_average = @eleven_point_precision.values.inject(0.0) { |sum, precision| sum + precision } / 11
367:         
368:       rescue
369:         raise "Error while calculating the 11-point precision map!"
370:       end
